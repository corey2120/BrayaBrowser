🚀 PUBLISH TO AUR (Copy-Paste Commands)
════════════════════════════════════════════════════════════════

# 1. Push your fixes & create tag
cd /home/cobrien/Documents/BrayaBrowser
git add src/BrayaTab.h
git commit -m "Fix: Add missing member variables"
git tag -a v1.0.6 -m "Release 1.0.6"
git push origin main --tags

# 2. Set up AUR (first time only)
# - Register: https://aur.archlinux.org/register
# - Add SSH key: cat ~/.ssh/id_ed25519.pub

# 3. Publish to AUR
mkdir -p ~/aur && cd ~/aur
git clone ssh://aur@aur.archlinux.org/braya-browser.git
cd braya-browser
cp /home/cobrien/Documents/BrayaBrowser/PKGBUILD .
cp /home/cobrien/Documents/BrayaBrowser/.SRCINFO .
git add PKGBUILD .SRCINFO
git commit -m "Initial import: BrayaBrowser 1.0.6"
git push

💾 TEST LOCALLY
════════════════════════════════════════════════════════════════

# Install the package
cd /home/cobrien/Documents/braya-browser-aur
sudo pacman -U braya-browser-1.0.6-1-x86_64.pkg.tar.zst

# Run browser

# Uninstall
